@app.get("/movies")
def get_all_movies(db:Session = Depends(get_session)):
    return db.query(MovieModel).all()

@app.get("/movie/{id}")
def get_movie(id:int,db:Session = Depends(get_session)):
    db_movie = db.query(MovieModel).filter(MovieModel.id==id).first()
    if db_movie is None:
        raise HTTPException(status_code=404,detail="No movie found")
    return db_movie

@app.get("/movie/title/{title}")
def get_movie(title:str,db:Session = Depends(get_session)):
    db_movie = db.query(MovieModel).filter(MovieModel.title == title.capitalize()).first()
    if db_movie is None:
        raise HTTPException(status_code=404,detail="Movie not found")
    return db_movie

def get_movies():
    try:
        resp = requests.get(url=f"{API_URL}/movies")
        resp.raise_for_status()
        resp_movies = resp.json()
        movies = [MovieModel(**movie) for movie in resp_movies]
        return movies
    except Exception as e:
        return []
